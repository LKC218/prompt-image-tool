## 三、T1 — 导航 Bug 修复：详细页编辑保存后无法返回首页（P0）

### 3.1 Bug 描述

**复现路径：** 首页 → 点击提示词模板 → 进入详细页 → 点击编辑 → 进入编辑器 → 保存 → 返回详细页 → 点击返回 → **卡住在详细页，无法返回首页**

**伴随现象：** 首页底部导航栏 UI 停留在详细页状态（导航栏显示异常）

### 3.2 根因分析

**核心问题：`goBack()` 中 `history.back()` 触发 `popstate` 事件，导致路由栈被双重弹出。**

当前 [mobile-router.js:55-67](file:///d:/Work/git/prompt-image-tool/src/js/mobile-router.js#L55-L67) 的 `goBack()` 实现：

```javascript
function goBack() {
    if (routeStack.length === 0) return false;
    const prevRoute = routeStack.pop();
    const currentSnapshot = currentRoute;
    currentRoute = prevRoute;

    history.back();  // ← 触发 popstate 事件

    if (onRouteChange) {
        onRouteChange(currentRoute, currentSnapshot, 'pop');
    }
    return true;
}
```

同时 [mobile-router.js:97-108](file:///d:/Work/git/prompt-image-tool/src/js/mobile-router.js#L97-L108) 注册了 `popstate` 监听：

```javascript
window.addEventListener('popstate', (e) => {
    if (!e.state || e.state.view !== 'mobile') return;
    if (routeStack.length > 0) {
        const prev = routeStack.pop();  // ← 第二次弹出！
        const currentSnapshot = currentRoute;
        currentRoute = prev;
        if (onRouteChange) {
            onRouteChange(currentRoute, currentSnapshot, 'pop');
        }
    }
});
```

**双重弹出流程：**

1. 编辑器保存 → `goBack()` → 弹出 routeStack（detail 路由）→ `currentRoute = detail`
2. `history.back()` → 触发 popstate → 再次弹出 routeStack（home 路由）→ `currentRoute = home`
3. `goBack()` 的 `onRouteChange` 回调执行 → 导航到 detail
4. popstate 的 `onRouteChange` 回调执行 → 导航到 home

结果：routeStack 被清空，用户在 detail 页点击返回时 routeStack 已空，`goBack()` 返回 `false`，无法返回。

**伴随问题：** `createPage()` 在 `direction='pop'` 时，对同一 routeKey 的旧页面未从 DOM 中移除，导致 DOM 中存在重复页面元素。

### 3.3 修复方案

#### F1.1 消除 popstate 双重弹出

**方案：在 `goBack()` 中设置标志位，popstate 处理器检测到标志位时跳过处理。**

```javascript
let _programmaticBack = false;

function goBack() {
    if (routeStack.length === 0) return false;
    const prevRoute = routeStack.pop();
    const currentSnapshot = currentRoute;
    currentRoute = prevRoute;

    _programmaticBack = true;
    history.back();

    if (onRouteChange) {
        onRouteChange(currentRoute, currentSnapshot, 'pop');
    }
    return true;
}

// popstate 处理器
window.addEventListener('popstate', (e) => {
    if (!e.state || e.state.view !== 'mobile') return;
    if (_programmaticBack) {
        _programmaticBack = false;
        return;  // 跳过，已由 goBack() 处理
    }
    // 仅处理浏览器前进/后退按钮触发的情况
    if (routeStack.length > 0) {
        const prev = routeStack.pop();
        const currentSnapshot = currentRoute;
        currentRoute = prev;
        if (onRouteChange) {
            onRouteChange(currentRoute, currentSnapshot, 'pop');
        }
    }
});
```

#### F1.2 修复 createPage 重复页面问题

在 `createPage()` 中，创建新页面前先移除同 routeKey 的旧页面：

```javascript
function createPage(routeKey, params = {}, direction = 'tab') {
    // 移除同 routeKey 的旧页面
    if (activePages[routeKey]) {
        const oldPage = activePages[routeKey];
        if (oldPage.handler && oldPage.handler.unmount) oldPage.handler.unmount(oldPage.el);
        if (oldPage.el.parentNode) oldPage.el.remove();
        delete activePages[routeKey];
    }
    // ... 后续创建逻辑不变
}
```

#### F2.1 提取图片工具为共享模块

```javascript
// src/js/image-utils.js
function readFileAsDataURL(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = () => resolve(reader.result);
        reader.onerror = () => reject(reader.error);
        reader.readAsDataURL(file);
    });
}

function compressImageToJpeg(dataUrl, quality = 0.92) {
    return new Promise((resolve) => {
        const img = new Image();
        img.onload = () => {
            const canvas = document.createElement('canvas');
            canvas.width = img.naturalWidth;
            canvas.height = img.naturalHeight;
            const ctx = canvas.getContext('2d');
            ctx.fillStyle = '#FFFFFF';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.drawImage(img, 0, 0);
            canvas.toBlob((blob) => {
                if (!blob) { resolve(dataUrl); return; }
                const originalSize = Math.round((dataUrl.length - dataUrl.indexOf(',') - 1) * 3 / 4);
                if (blob.size >= originalSize) { resolve(dataUrl); return; }
                const reader = new FileReader();
                reader.onloadend = () => resolve(reader.result);
                reader.readAsDataURL(blob);
            }, 'image/jpeg', quality);
        };
        img.onerror = () => resolve(dataUrl);
        img.src = dataUrl;
    });
}

export { readFileAsDataURL, compressImageToJpeg };
```

PC 端 `app.js` 改为从 `image-utils.js` 导入，保持功能不变。

#### F5.3 过滤逻辑改造

```javascript
function getFilteredList() {
    if (!libraryData) return [];
    let list = [...libraryData.promptSets];

    if (searchKeyword) {
        const kw = searchKeyword.toLowerCase();
        list = list.filter(p => p.name.toLowerCase().includes(kw));
    }

    if (currentFilter === 'all') {
        list.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    } else if (currentFilter === 'favorite') {
        list = list.filter(p => p.isFavorite);
    } else if (currentFilter === 'recent') {
        list.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    } else if (currentFilter.startsWith('tag:')) {
        const tagName = currentFilter.slice(4);
        list = list.filter(p => {
            let tags = [];
            try { tags = JSON.parse(p.tags || '[]'); } catch (e) { tags = []; }
            return tags.includes(tagName);
        });
    }

    return list;
}
```
